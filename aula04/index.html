        import math

        def resolver_equacao():
            print("--- Calculadora de Equação do 2º Grau ---")
            
            # Entrada dos coeficientes
            a = float(input("Digite o valor de a: "))
            
            if a == 0:
                print("O valor de 'a' não pode ser zero em uma equação do 2º grau.")
                return

            b = float(input("Digite o valor de b: "))
            c = float(input("Digite o valor de c: "))

            # Cálculo do Delta (Δ)
            delta = b**2 - 4*a*c
            print(f"\nDelta (Δ) = {delta}")

            # Verificação das raízes
            if delta < 0:
                print("A equação não possui raízes reais (delta negativo).")
            elif delta == 0:
                x = -b / (2*a)
                print(f"A equação possui uma única raiz real: x = {x}")
            else:
                x1 = (-b + math.sqrt(delta)) / (2*a)
                x2 = (-b - math.sqrt(delta)) / (2*a)
                print(f"Raiz x1 = {x1}")
                print(f"Raiz x2 = {x2}")

        # Executa a função
        resolver_equacao()
